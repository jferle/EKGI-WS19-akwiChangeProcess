
Attribut einer Person ausgeben

PREFIX schema: <https://schema.org/>
PREFIX akwi: <https://bmake.th-brandenburg.de/akwi/>
Select *
Where{
  akwi:albus_dumbledore schema:familyName ?a;
	schema:givenName ?b;
    schema:honorificPrefix ?c;
    schema:telephone ?e;
    schema:eMail ?f;
    ^schema:employee ?g.                                                     
}

	  schema:familyName "Dumbledore" ;

	  schema:telephone "0874324839582" ;
  	  schema:eMail "albus.dumbledore@hogwarts.ch" .

}
