

-- Fakultät von Hochschule anzeigen

PREFIX schema: <https://schema.org/>
PREFIX akwi:  <https://bmake.th-brandenburg.de/akwi/>
Select ?1
Where{
  akwi:hogwarts_information_technology schema:name ?1.
}

-- Fakultät von Hochschule hinzufügen

PREFIX schema: <https://schema.org/>
PREFIX akwi:  <https://bmake.th-brandenburg.de/akwi/>
Insert{
  akwi:hogwarts schema:department akwi:hogwarts_information_technology.
  akwi:hogwarts_information_technology a akwi:DepartmentOrFaculty;
                                       schema:name "Information Technology".
}
Where{}

-- Fakultät von Hochschule löschen

PREFIX schema: <https://schema.org/>
PREFIX akwi:  <https://bmake.th-brandenburg.de/akwi/>
Delete{
  akwi:hogwarts schema:department akwi:hogwarts_information_technology.
  akwi:hogwarts_information_technology a akwi:DepartmentOrFaculty;
                                       schema:name ?1.
}
Where{
  akwi:hogwarts schema:department akwi:hogwarts_information_technology.
  akwi:hogwarts_information_technology a akwi:DepartmentOrFaculty;
  				       schema:name ?1.
}
