
-- eine bestimmte Person auswählen

PREFIX schema: <https://schema.org/>
Select ?person ?a ?b ?c ?d ?e
Where{
  ?person schema:familyName ?a ;
          schema:givenName ?b ;
          schema:telephone ?c ;
          schema:honorificPrefix ?d ;
          schema:eMail ?e ;
          schema:familyName "Schmidtmann";
          schema:givenName "Achim" .
}
